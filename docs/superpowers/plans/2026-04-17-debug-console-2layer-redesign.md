# Debug Console 2-Layer Redesign Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Replace the flat debug-console command surface in `gol-project` with a disciplined 2-layer (`cmd + category + target + params`) model backed by a modular, typed-param registry — shrinking `service_console.gd` from 971 lines to ~150 and eliminating per-command completion branches in `console_panel.gd`.

**Architecture:** New `scripts/debug/console/` package with declarative `CommandSpec` / `SubcommandSpec` / `ParamSpec` data objects, a reflection-free dispatcher in `console_registry.gd`, a typed completion provider keyed by `ParamType`, and one module file per command category. `service_console.gd` becomes a thin lifecycle shell that delegates `execute()` / completion queries to the registry. Clean break — no name aliases; typo hints (Levenshtein ≤ 2) soften discovery.

**Tech Stack:** Godot 4.6, GDScript, GECS ECS, gdUnit4 (unit tests), SceneConfig (integration tests).

**Reference:** Design spec lives at `docs/superpowers/specs/2026-04-17-debug-console-2layer-redesign-design.md` — read the "Command Surface" table, "Data Model", and "Dispatch Flow" sections before starting.

```gdscript
# console dispatcher. Each value must have a matching branch in

extends RefCounted

	STRING,       # raw string, no validation or completion
	INT,          # parsed with is_valid_int / to_int

	EXPRESSION,   # trailing-slurp string (consumes all remaining tokens)
}
```

```gdscript
# CommandSpec / SubcommandSpec / ParamSpec — immutable declarative command
# descriptors consumed by ConsoleRegistry. Modules build these from their
# build() method; the registry never inspects handlers by reflection.
class_name ConsoleCommandSpec
extends RefCounted


class ParamSpec extends RefCounted:
	var name: String = ""
	var type: int = 0
	var required: bool = false
	var default: Variant = null
	var enum_values: Array = []

	static func required(p_name: String, p_type: int) -> ParamSpec:
		var p := ParamSpec.new()
		p.name = p_name
		p.type = p_type
		p.required = true
		return p

	static func optional(p_name: String, p_type: int, p_default: Variant) -> ParamSpec:
		var p := ParamSpec.new()
		p.name = p_name
		p.type = p_type
		p.required = false
		p.default = p_default
		return p

	static func enum_required(p_name: String, values: Array) -> ParamSpec:
		var p := ParamSpec.new()
		p.name = p_name
		p.type = 4  # ConsoleParamType.ENUM
		p.required = true
		p.enum_values = values
		return p


class SubcommandSpec extends RefCounted:
	var name: String = ""
	var desc: String = ""
	var params: Array = []  # Array[ParamSpec]
	var handler: Callable = Callable()

	func _init(p_name: String = "", p_desc: String = "", p_params: Array = [], p_handler: Callable = Callable()) -> void:
		name = p_name
		desc = p_desc
		params = p_params
		handler = p_handler


class CommandSpec extends RefCounted:
	var name: String = ""
	var desc: String = ""
	var subcommands: Array = []     # Array[SubcommandSpec]; empty → flat
	var flat_params: Array = []     # Array[ParamSpec]; only when subcommands empty
	var flat_handler: Callable = Callable()
	var default_subcommand: String = ""

	static func category(p_name: String, p_desc: String, p_subs: Array, p_default: String = "") -> CommandSpec:
		var c := CommandSpec.new()
		c.name = p_name
		c.desc = p_desc
		c.subcommands = p_subs
		c.default_subcommand = p_default
		return c

	static func flat(p_name: String, p_desc: String, p_params: Array, p_handler: Callable) -> CommandSpec:
		var c := CommandSpec.new()
		c.name = p_name
		c.desc = p_desc
		c.flat_params = p_params
		c.flat_handler = p_handler
		return c

	func is_flat() -> bool:
		return subcommands.is_empty()

	func find_subcommand(sub_name: String) -> SubcommandSpec:
		for s in subcommands:
			if (s as SubcommandSpec).name == sub_name:
				return s
		return null
```

```gdscript
# ConsoleHelpers — stateless helpers used by console command modules.
# Moved verbatim from service_console.gd. All methods are static.
class_name ConsoleHelpers
extends RefCounted

const SPAWN_GRID_SPACING: float = 24.0
const LOOT_BOX_TEXTURE_PATH: String = "res://assets/sprite_sheets/boxes/box_re_texture.png"


static func find_player() -> Entity:
	if not ECS.world:
		return null
	for entity in ECS.world.entities:
		if is_instance_valid(entity) and entity.has_component(CPlayer):
			var camp: CCamp = entity.get_component(CCamp)
			if camp and camp.camp == CCamp.CampType.PLAYER:
				return entity
	return null


static func find_campfire() -> Entity:
	if not ECS.world:
		return null
	for entity in ECS.world.entities:
		if is_instance_valid(entity) and entity.has_component(CCampfire):
			if not entity.has_component(CDead):
				return entity
	return null


static func find_target(target_filter: String) -> Array:
	if not ECS.world:
		return [null, ""]
	var lower := target_filter.to_lower()
	if lower == "player":
		return [find_player(), "Player"]
	if lower == "campfire":
		return [find_campfire(), "Campfire"]
	for entity in ECS.world.entities:
		if not is_instance_valid(entity):
			continue
		var entity_name := get_entity_display_name(entity)
		if entity_name.to_lower().find(lower) != -1:
			return [entity, entity_name]
	return [null, ""]


static func get_entity_display_name(entity: Entity) -> String:
	if not is_instance_valid(entity):
		return "(freed)"
	var node_name := String(entity.name)
	if entity.has_component(CPlayer):
		return "Player_" + node_name
	if entity.has_component(CGuard):
		return "Guard_" + node_name
	if entity.has_component(CCampfire):
		return "Campfire_" + node_name
	if entity.has_component(CBullet):
		return "Bullet_" + node_name
	if entity.has_component(CGoapAgent):
		var camp_comp: CCamp = entity.get_component(CCamp)
		if camp_comp and camp_comp.camp == CCamp.CampType.ENEMY:
			return "Enemy_" + node_name
		return "AI_" + node_name
	return node_name


static func matching_recipe_ids(filter: String = "") -> Array[String]:
	var recipe_service := ServiceContext.recipe()
	if recipe_service == null:
		return []
	var recipe_ids: Array[String] = []
	for recipe_id in recipe_service.get_all_recipe_ids():
		recipe_ids.append(recipe_id)
	if filter.is_empty():
		recipe_ids.sort()
		return recipe_ids
	var matches: Array[String] = []
	var filter_lower := filter.to_lower()
	for recipe_id in recipe_ids:
		if recipe_id.to_lower().find(filter_lower) != -1:
			matches.append(recipe_id)
	matches.sort()
	return matches


static func resolve_spawn_origin(x: Variant, y: Variant) -> Dictionary:
	var x_empty := (x == null) or (x is String and (x as String).is_empty())
	var y_empty := (y == null) or (y is String and (y as String).is_empty())

	if x_empty and y_empty:
		var player := find_player()
		if not player:
			return {"error": "Error: Player not found"}
		var player_transform: CTransform = player.get_component(CTransform)
		if not player_transform:
			return {"error": "Error: Player has no transform"}
		return {"position": player_transform.position}

	if x_empty or y_empty:
		return {"error": "Usage: spawn entity <recipe_id> [count=1] [x] [y]"}

	var fx := _to_float_or_nan(x)
	var fy := _to_float_or_nan(y)
	if is_nan(fx) or is_nan(fy):
		return {"error": "Invalid coordinates"}

	return {"position": Vector2(fx, fy)}


static func resolve_spawn_origin_mouse(root: Node, x: Variant, y: Variant) -> Dictionary:
	var x_empty := (x == null) or (x is String and (x as String).is_empty())
	var y_empty := (y == null) or (y is String and (y as String).is_empty())

	if x_empty and y_empty:
		if root:
			var viewport := root.get_viewport()
			if viewport:
				var canvas_transform := viewport.get_canvas_transform()
				var screen_pos: Vector2 = viewport.get_mouse_position()
				var world_pos: Vector2 = canvas_transform.affine_inverse() * screen_pos
				return {"position": world_pos}
		var player := find_player()
		if not player:
			return {"error": "Error: Player not found and mouse position unavailable"}
		var player_transform: CTransform = player.get_component(CTransform)
		if not player_transform:
			return {"error": "Error: Player has no transform"}
		return {"position": player_transform.position}

	if x_empty or y_empty:
		return {"error": "Usage: spawn box <recipe_id> [x] [y]"}

	var fx := _to_float_or_nan(x)
	var fy := _to_float_or_nan(y)
	if is_nan(fx) or is_nan(fy):
		return {"error": "Invalid coordinates"}

	return {"position": Vector2(fx, fy)}


static func spawn_offset(index: int) -> Vector2:
	var directions: Array[Vector2] = [
		Vector2.RIGHT, Vector2.LEFT, Vector2.UP, Vector2.DOWN,
		Vector2(1.0, -1.0), Vector2(-1.0, -1.0), Vector2(1.0, 1.0), Vector2(-1.0, 1.0),
	]
	var ring := int(floor(float(index) / float(directions.size()))) + 1
	var direction: Vector2 = directions[index % directions.size()]
	return direction * SPAWN_GRID_SPACING * float(ring)


static func instantiate_component(class_name_str: String) -> Component:
	for entry: Dictionary in ProjectSettings.get_global_class_list():
		if entry.get("class", "") != class_name_str:
			continue
		var path: String = entry.get("path", "")
		if path.is_empty():
			continue
		var script = load(path) as GDScript
		if script == null or not script.can_instantiate():
			continue
		var instance = script.new()
		if instance is Component:
			return instance
	return null


static func component_class_names() -> Array[String]:
	var names: Array[String] = []
	for entry: Dictionary in ProjectSettings.get_global_class_list():
		var path: String = entry.get("path", "")
		if path.begins_with("res://scripts/components/"):
			names.append(entry.get("class", ""))
	names.sort()
	return names


static func _to_float_or_nan(v: Variant) -> float:
	if v is float:
		return v
	if v is int:
		return float(v)
	if v is String:
		var s: String = v
		if s.is_valid_float():
			return s.to_float()
	return NAN
```

```gdscript
# ConsoleParser — tokenisation + cursor-slot detection.
# Stateless; all methods static. Input is already lowercased by callers
# that need case-insensitive matching (the dispatcher does this).
class_name ConsoleParser
extends RefCounted


# parse() produces the structure consumed by the dispatcher:
#   { cmd: String, subcmd_candidate: String, positionals: Array[String] }
# subcmd_candidate is the raw second token; the registry decides whether
# it's a valid subcommand or should be treated as the first positional of
# a flat command.
static func parse(input: String) -> Dictionary:
	var parts := input.strip_edges().split(" ", false)
	if parts.is_empty():
		return {"cmd": "", "subcmd_candidate": "", "positionals": []}
	var cmd: String = parts[0]
	var subcmd: String = parts[1] if parts.size() > 1 else ""
	var positionals: Array = []
	if parts.size() > 2:
		positionals = Array(parts.slice(2))
	return {"cmd": cmd, "subcmd_candidate": subcmd, "positionals": positionals}


# describe_cursor() returns a slot descriptor for the completion flow:
#   slot = "cmd":        { slot, prefix }
#   slot = "subcmd":     { slot, cmd, prefix }
#   slot = "positional": { slot, cmd, subcmd, param_index, prefix }
# positional.subcmd is "" when the command turned out to be flat — the
# completer resolves final flat-vs-category decision via the registry.
static func describe_cursor(input: String, cursor: int) -> Dictionary:
	var left := input.substr(0, clampi(cursor, 0, input.length()))
	# Count complete tokens (split by whitespace, non-empty).
	var trailing_space := left.ends_with(" ")
	var tokens := Array(left.strip_edges(false, true).split(" ", false))
	var in_progress := ""
	if not trailing_space and tokens.size() > 0:
		in_progress = tokens.pop_back()
	# tokens now contains the fully-typed tokens before cursor.
	# in_progress is the partial token at cursor (possibly "").

	var token_count := tokens.size()
	if token_count == 0:
		return {"slot": "cmd", "prefix": in_progress}

	if token_count == 1:
		return {"slot": "subcmd", "cmd": tokens[0], "prefix": in_progress}

	# token_count >= 2 → positional slot.
	# param_index is the count of positional tokens already typed (excluding in_progress).
	return {
		"slot": "positional",
		"cmd": tokens[0],
		"subcmd": tokens[1],
		"param_index": token_count - 2,
		"prefix": in_progress,
	}
```

```gdscript
# ConsoleCompleter — maps (ParamType, prefix) → Array[String] of candidates.
# All branches are stateless; RECIPE / COMPONENT / ENTITY branches read from
# ConsoleHelpers at call time. Missing services return [] rather than errors.
class_name ConsoleCompleter
extends RefCounted

const ParamTypeScript := preload("res://scripts/debug/console/console_types.gd")
const ConsoleHelpersScript := preload("res://scripts/debug/console/console_helpers.gd")


static func for_param_type(param_type: int, prefix: String, enum_values: Array) -> Array[String]:
	match param_type:
		ParamTypeScript.RECIPE:
			return ConsoleHelpersScript.matching_recipe_ids(prefix)
		ParamTypeScript.COMPONENT:
			return _filter_prefix(ConsoleHelpersScript.component_class_names(), prefix)
		ParamTypeScript.BOOL_TOGGLE:
			return _filter_prefix(["off", "on"], prefix)
		ParamTypeScript.ENUM:
			var sorted := enum_values.duplicate()
			sorted.sort()
			var typed: Array[String] = []
			for v in sorted:
				typed.append(String(v))
			return _filter_prefix(typed, prefix)
		ParamTypeScript.ENTITY:
			return _entity_names_matching(prefix)
		_:
			return [] as Array[String]


static func _filter_prefix(sorted_source: Array, prefix: String) -> Array[String]:
	var results: Array[String] = []
	var lower := prefix.to_lower()
	for item in sorted_source:
		var s := String(item)
		if lower.is_empty() or s.to_lower().begins_with(lower):
			results.append(s)
	return results


static func _entity_names_matching(prefix: String) -> Array[String]:
	var results: Array[String] = []
	if not ECS.world:
		return results
	var lower := prefix.to_lower()
	var seen := {}
	for entity in ECS.world.entities:
		if not is_instance_valid(entity):
			continue
		var name := ConsoleHelpersScript.get_entity_display_name(entity)
		if seen.has(name):
			continue
		if lower.is_empty() or name.to_lower().begins_with(lower):
			results.append(name)
			seen[name] = true
	results.sort()
	return results
```

```gdscript
# completion / help queries. Created per Service_Console instance.
class_name ConsoleRegistry
extends RefCounted

const ConsoleParserScript := preload("res://scripts/debug/console/console_parser.gd")
const ConsoleCompleterScript := preload("res://scripts/debug/console/console_completer.gd")
const ConsoleContextScript := preload("res://scripts/debug/console/console_context.gd")
const CommandSpecScript := preload("res://scripts/debug/console/console_command.gd")
const ParamTypeScript := preload("res://scripts/debug/console/console_types.gd")

const _MODULES: Array = [
	preload("res://scripts/debug/console/commands/spawn_command.gd"),
	preload("res://scripts/debug/console/commands/add_command.gd"),
	preload("res://scripts/debug/console/commands/remove_command.gd"),
	preload("res://scripts/debug/console/commands/damage_command.gd"),
	preload("res://scripts/debug/console/commands/time_command.gd"),
	preload("res://scripts/debug/console/commands/refresh_command.gd"),
	preload("res://scripts/debug/console/commands/basic_commands.gd"),
]

var _commands: Dictionary = {}  # String → CommandSpec
var _sorted_names: Array[String] = []
var _service: Object  # Service_Console back-ref


func _init(service: Object) -> void:
	_service = service


func setup() -> void:
	for module_script in _MODULES:
		var module = module_script.new()
		var specs: Array = module.build()
		for spec in specs:
			_register_spec(spec)
	_finalize_setup()


func teardown() -> void:
	_commands.clear()
	_sorted_names.clear()


# Exposed for unit tests that bypass module discovery.
func _register_spec(spec) -> void:
	_commands[String(spec.name)] = spec


func _finalize_setup() -> void:
	_sorted_names.clear()
	for n in _commands.keys():
		_sorted_names.append(String(n))
	_sorted_names.sort()


# --- Public API used by Service_Console ---

func command_names() -> Array[String]:
	return _sorted_names.duplicate()


func has_command(name: String) -> bool:
	return _commands.has(name)


func execute(input: String) -> String:
	var parsed: Dictionary = ConsoleParserScript.parse(input)
	var cmd_name: String = String(parsed.cmd).to_lower()
	if cmd_name.is_empty():
		return ""

	if not _commands.has(cmd_name):
		return _unknown_cmd_message(cmd_name)

	var spec = _commands[cmd_name]
	if spec.is_flat():
		return _invoke_flat(spec, parsed.subcmd_candidate, parsed.positionals)
	return _invoke_category(spec, parsed.subcmd_candidate, parsed.positionals)


func describe_cursor(input: String, cursor: int) -> Dictionary:
	return ConsoleParserScript.describe_cursor(input, cursor)


func get_completions_at(state: Dictionary) -> Array[String]:

		"cmd":
			return _cmd_completions(state.get("prefix", ""))
		"subcmd":
			return _subcmd_completions(state.get("cmd", ""), state.get("prefix", ""))
		"positional":
			return _positional_completions(state)
		_:
			return [] as Array[String]


func help_for(name: String) -> String:
	if name.is_empty():
		var lines: Array[String] = ["Available commands:"]
		for n in _sorted_names:
			var spec = _commands[n]
			if spec.desc.is_empty():
				lines.append("  " + n)
			else:
				lines.append("  %s - %s" % [n, spec.desc])
		return "\n".join(lines)

	if not _commands.has(name):
		return "Unknown command: " + name
	return _build_usage(_commands[name])


# --- Resolution + dispatch ---

func _invoke_flat(spec, second_token: String, positionals: Array) -> String:
	var all_positionals: Array = []
	if not second_token.is_empty():
		all_positionals.append(second_token)
	all_positionals.append_array(positionals)
	return _bind_and_call(spec.flat_handler, spec.flat_params, all_positionals, _usage_prefix_flat(spec))


func _invoke_category(spec, subcmd_name: String, positionals: Array) -> String:
	if subcmd_name.is_empty():
		if not spec.default_subcommand.is_empty():
			var default_sub = spec.find_subcommand(spec.default_subcommand)
			if default_sub != null:
				return _bind_and_call(default_sub.handler, default_sub.params, positionals, _usage_prefix_sub(spec, default_sub))
		return _subcommand_list_message(spec)

	var sub = spec.find_subcommand(subcmd_name.to_lower())
	if sub == null:
		return _unknown_subcmd_message(spec, subcmd_name)

	return _bind_and_call(sub.handler, sub.params, positionals, _usage_prefix_sub(spec, sub))


func _bind_and_call(handler: Callable, param_specs: Array, positionals: Array, usage_prefix: String) -> String:
	var args := {}
	for i in range(param_specs.size()):
		var p = param_specs[i]
		if i < positionals.size():
			var token: String = String(positionals[i])
			# EXPRESSION slurps all remaining tokens.
			if p.type == ParamTypeScript.EXPRESSION:
				token = " ".join(positionals.slice(i))
			var coerced = _coerce(p, token)
			if coerced.has("error"):
				return coerced.error
			args[p.name] = coerced.value
		else:
			if p.required:
				return "Usage: %s%s" % [usage_prefix, _format_params(param_specs)]
			args[p.name] = p.default
	var ctx := ConsoleContextScript.new(_service)
	return handler.call(ctx, args)


func _coerce(param, token: String) -> Dictionary:
	match param.type:
		ParamTypeScript.INT:
			if not token.is_valid_int():
				return {"error": "Invalid %s: '%s' (expected int)" % [param.name, token]}
			return {"value": token.to_int()}
		ParamTypeScript.FLOAT:
			if not token.is_valid_float():
				return {"error": "Invalid %s: '%s' (expected float)" % [param.name, token]}
			return {"value": token.to_float()}
		ParamTypeScript.BOOL_TOGGLE:
			var low := token.to_lower()
			if low in ["on", "true", "1"]:
				return {"value": true}
			if low in ["off", "false", "0"]:
				return {"value": false}
			return {"error": "Invalid %s: '%s' (expected on/off)" % [param.name, token]}
		ParamTypeScript.ENUM:
			if not (token in param.enum_values):
				return {"error": "Invalid %s: '%s' (expected one of %s)" % [param.name, token, ", ".join(param.enum_values)]}
			return {"value": token}
		_:
			return {"value": token}


# --- Messages ---

func _unknown_cmd_message(cmd: String) -> String:
	var suggestion := _nearest(cmd, _sorted_names)
	if suggestion.is_empty():
		return "Unknown command: '%s'. Type 'help' for available commands." % cmd
	return "Unknown command: '%s'. Did you mean '%s'?" % [cmd, suggestion]


func _unknown_subcmd_message(spec, sub: String) -> String:
	var names: Array[String] = []
	for s in spec.subcommands:
		names.append(s.name)
	names.sort()
	var suggestion := _nearest(sub, names)
	var head := "Unknown subcommand: '%s %s'." % [spec.name, sub]
	if not suggestion.is_empty():
		return "%s Did you mean '%s %s'?" % [head, spec.name, suggestion]
	return "%s Available: %s" % [head, ", ".join(names)]


func _subcommand_list_message(spec) -> String:
	var names: Array[String] = []
	for s in spec.subcommands:
		names.append(s.name)
	names.sort()
	return "Usage: %s <%s>" % [spec.name, "|".join(names)]


func _usage_prefix_flat(spec) -> String:
	return spec.name + " "


func _usage_prefix_sub(spec, sub) -> String:
	return "%s %s " % [spec.name, sub.name]


func _format_params(param_specs: Array) -> String:
	var parts: Array[String] = []
	for p in param_specs:
		if p.required:
			parts.append("<%s>" % p.name)
		elif p.default == null:
			parts.append("[%s]" % p.name)
		else:
			parts.append("[%s=%s]" % [p.name, str(p.default)])
	return " ".join(parts)


func _build_usage(spec) -> String:
	if spec.is_flat():
		var line := "%s %s" % [spec.name, _format_params(spec.flat_params)]
		if spec.desc.is_empty():
			return line
		return "%s\n  %s" % [line, spec.desc]
	var lines: Array[String] = []
	for sub in spec.subcommands:
		var l := "%s %s %s" % [spec.name, sub.name, _format_params(sub.params)]
		if sub.desc.is_empty():
			lines.append(l)
		else:
			lines.append("%s\n  %s" % [l, sub.desc])
	return "\n".join(lines)


# --- Completion ---

func _cmd_completions(prefix: String) -> Array[String]:
	var lower := prefix.to_lower()
	var results: Array[String] = []
	for n in _sorted_names:
		if lower.is_empty() or n.begins_with(lower):
			results.append(n)
	return results


func _subcmd_completions(cmd: String, prefix: String) -> Array[String]:
	if not _commands.has(cmd):
		return [] as Array[String]
	var spec = _commands[cmd]
	if spec.is_flat():
		# For flat commands, second slot is param[0].
		return _param_completions(spec.flat_params, 0, prefix)
	var results: Array[String] = []
	var lower := prefix.to_lower()
	for s in spec.subcommands:
		if lower.is_empty() or s.name.begins_with(lower):
			results.append(s.name)
	results.sort()
	return results


func _positional_completions(state: Dictionary) -> Array[String]:
	var cmd_name := String(state.get("cmd", ""))
	if not _commands.has(cmd_name):
		return [] as Array[String]
	var spec = _commands[cmd_name]
	var prefix := String(state.get("prefix", ""))
	var param_index := int(state.get("param_index", 0))
	if spec.is_flat():
		# Flat: state.subcmd is really the first positional; param_index is offset by 1.
		return _param_completions(spec.flat_params, param_index + 1, prefix)
	var sub = spec.find_subcommand(String(state.get("subcmd", "")).to_lower())
	if sub == null:
		return [] as Array[String]
	return _param_completions(sub.params, param_index, prefix)


func _param_completions(param_specs: Array, index: int, prefix: String) -> Array[String]:
	if index < 0 or index >= param_specs.size():
		return [] as Array[String]
	var p = param_specs[index]
	return ConsoleCompleterScript.for_param_type(p.type, prefix, p.enum_values)


# --- Typo hints (Levenshtein) ---

func _nearest(target: String, candidates: Array) -> String:
	var best := ""
	var best_dist := 3  # require distance ≤ 2
	for c in candidates:
		var d := _levenshtein(target, String(c))
		if d < best_dist:
			best_dist = d
			best = String(c)
	return best


static func _levenshtein(a: String, b: String) -> int:
	var la := a.length()
	var lb := b.length()
	if la == 0:
		return lb
	if lb == 0:
		return la
	var prev := []
	prev.resize(lb + 1)
	for j in range(lb + 1):
		prev[j] = j
	for i in range(1, la + 1):
		var curr := []
		curr.resize(lb + 1)
		curr[0] = i
		for j in range(1, lb + 1):
			var cost := 0 if a[i - 1] == b[j - 1] else 1
			curr[j] = min(
				min(curr[j - 1] + 1, prev[j] + 1),
				prev[j - 1] + cost
			)
		prev = curr
	return prev[lb]
```

```gdscript
	return [
		CommandSpecScript.CommandSpec.category("add", "Add things to entities", [
			CommandSpecScript.SubcommandSpec.new(
				"comp",
				"Attach a component to a target (default: player)",
				[
					CommandSpecScript.ParamSpec.required("class_name", ParamTypeScript.COMPONENT),
					CommandSpecScript.ParamSpec.optional("target", ParamTypeScript.ENTITY, "player"),
				],
				Callable(self, "_add_comp")

		])

func _add_comp(ctx, args: Dictionary) -> String:
	var class_name_str: String = String(args.class_name)
	var target_filter: String = String(args.target)

	var comp = ctx.instantiate_component(class_name_str)
	if comp == null:
		var available := ctx.component_class_names()
		return "Unknown component: %s\nAvailable:\n  %s" % [class_name_str, "\n  ".join(available)]

	var resolved: Array = ctx.find_target(target_filter)
	var target: Entity = resolved[0]
	var target_name: String = resolved[1]
	if not target:
		return "Error: Target not found: " + target_filter

	if target.has_component(comp.get_script()):
		var existing = target.get_component(comp.get_script())
		if existing.has_method("on_merge"):
			existing.on_merge(comp)
			return "Merged %s into %s" % [class_name_str, target_name]
		return "%s already has %s" % [target_name, class_name_str]

	target.add_component(comp)
```

```gdscript
class_name DamageCommand
extends ConsoleCommandModule

const CommandSpecScript := preload("res://scripts/debug/console/console_command.gd")
const ParamTypeScript := preload("res://scripts/debug/console/console_types.gd")

const DEFAULT_MULT: float = 1.0
const DEFAULT_INVINCIBLE: bool = false
const DEFAULT_MELEE_DAMAGE: float = 10.0


func build() -> Array:
	return [
		CommandSpecScript.CommandSpec.category("damage", "Damage debugging", [
			CommandSpecScript.SubcommandSpec.new(
				"deal",
				"Deal damage to a target",
				[
					CommandSpecScript.ParamSpec.required("amount", ParamTypeScript.FLOAT),
					CommandSpecScript.ParamSpec.optional("target", ParamTypeScript.ENTITY, "player"),
				],
				Callable(self, "_deal")
			),
			CommandSpecScript.SubcommandSpec.new(
				"weapon",
				"Set player melee weapon damage",
				[CommandSpecScript.ParamSpec.required("amount", ParamTypeScript.FLOAT)],
				Callable(self, "_weapon")
			),
			CommandSpecScript.SubcommandSpec.new(
				"mult",
				"Set (or show) player damage multiplier",
				[CommandSpecScript.ParamSpec.optional("value", ParamTypeScript.FLOAT, null)],
				Callable(self, "_mult")
			),
			CommandSpecScript.SubcommandSpec.new(
				"inv",
				"Toggle or set player invincibility (on/off)",
				[CommandSpecScript.ParamSpec.optional("state", ParamTypeScript.BOOL_TOGGLE, null)],
				Callable(self, "_inv")
			),
			CommandSpecScript.SubcommandSpec.new(
				"reset",
				"Reset all damage debug settings",
				[],
				Callable(self, "_reset")
			),
			CommandSpecScript.SubcommandSpec.new(
				"show",
				"Show damage info for a target",
				[CommandSpecScript.ParamSpec.optional("target", ParamTypeScript.ENTITY, "player")],
				Callable(self, "_show")
			),
		])
	]


func _deal(ctx, args: Dictionary) -> String:
	var amount: float = float(args.amount)
	if amount <= 0:
		return "Damage amount must be positive"
	var resolved := ctx.find_target(String(args.target))
	var target: Entity = resolved[0]
	var target_name: String = resolved[1]
	if not target:
		return "Error: Target not found: " + String(args.target)
	var hp: CHP = target.get_component(CHP)
	if not hp:
		return "Error: Target has no HP component"
	if hp.invincible_time > 0:
		return "Target is invincible (%.2fs remaining)" % hp.invincible_time
	var damage := CDamage.new()
	damage.amount = amount
	damage.knockback_direction = Vector2.ZERO
	target.add_component(damage)
	return "Dealt %.1f damage to %s (HP: %.1f/%.1f)" % [amount, target_name, hp.hp, hp.max_hp]


func _weapon(ctx, args: Dictionary) -> String:
	var amount: float = float(args.amount)
	if amount < 0:
		return "Damage amount cannot be negative"
	var player := ctx.find_player()
	if not player:
		return "Error: Player not found"
	var melee: CMelee = player.get_component(CMelee)
	if not melee:
		return "Error: Player has no melee weapon"
	var old := melee.damage
	melee.damage = amount
	return "Player melee damage: %.1f -> %.1f" % [old, amount]


func _mult(ctx, args: Dictionary) -> String:
	if args.value == null:
		return "Current damage multiplier: %.2fx" % ctx.service.get_player_damage_multiplier()
	var new_val: float = float(args.value)
	if new_val < 0:
		return "Multiplier cannot be negative"
	var old := ctx.service.get_player_damage_multiplier()
	ctx.service.set_player_damage_multiplier(new_val)
	return "Damage multiplier: %.2fx -> %.2fx" % [old, new_val]


func _inv(ctx, args: Dictionary) -> String:
	if args.state == null:
		var toggled := not ctx.service.is_player_invincible()
		ctx.service.set_player_invincible(toggled)
		return "Player invincibility: %s" % ("ON" if toggled else "OFF")
	ctx.service.set_player_invincible(bool(args.state))
	return "Player invincibility: %s" % ("ON" if args.state else "OFF")


func _reset(ctx, _args: Dictionary) -> String:
	var results: Array[String] = []
	if not is_equal_approx(ctx.service.get_player_damage_multiplier(), DEFAULT_MULT):
		ctx.service.set_player_damage_multiplier(DEFAULT_MULT)
		results.append("Multiplier: 1.0x")
	if ctx.service.is_player_invincible() != DEFAULT_INVINCIBLE:
		ctx.service.set_player_invincible(DEFAULT_INVINCIBLE)
		results.append("Invincibility: OFF")
	var player := ctx.find_player()
	if player:
		var melee: CMelee = player.get_component(CMelee)
		if melee and not is_equal_approx(melee.damage, DEFAULT_MELEE_DAMAGE):
			melee.damage = DEFAULT_MELEE_DAMAGE
			results.append("Melee Damage: %.1f" % DEFAULT_MELEE_DAMAGE)
	if results.is_empty():
		return "All damage settings already at defaults"
	return "Reset damage settings:\n  " + "\n  ".join(results)


func _show(ctx, args: Dictionary) -> String:
	var target_filter: String = String(args.target)
	var resolved := ctx.find_target(target_filter)
	var target: Entity = resolved[0]
	var target_name: String = resolved[1]
	if not target:
		return "Error: Target not found: " + target_filter
	var lines: Array[String] = ["=== Damage Info: %s ===" % target_name]
	var hp: CHP = target.get_component(CHP)
	if hp:
		lines.append("HP: %.1f / %.1f" % [hp.hp, hp.max_hp])
		if hp.invincible_time > 0:
			lines.append("Invincible: %.2fs remaining" % hp.invincible_time)
	else:
		lines.append("No HP component")
	var melee: CMelee = target.get_component(CMelee)
	if melee:
		lines.append("Melee Damage: %.1f" % melee.damage)
		lines.append("Attack Range: %.1f" % melee.attack_range)
		lines.append("Attack Interval: %.2fs" % melee.attack_interval)
		if melee.cooldown_remaining > 0:
			lines.append("Cooldown: %.2fs remaining" % melee.cooldown_remaining)
	if target_filter.to_lower() == "player":
		lines.append("Damage Multiplier: %.2fx" % ctx.service.get_player_damage_multiplier())
		lines.append("Invincibility: %s" % ("ON" if ctx.service.is_player_invincible() else "OFF"))
		if melee:
			lines.append("Effective Damage: %.1f" % (melee.damage * ctx.service.get_player_damage_multiplier()))
	var pending: CDamage = target.get_component(CDamage)
	if pending:
		lines.append("Pending Damage: %.1f" % pending.amount)
	if target.has_component(CDead):
		lines.append("Status: DEAD")
	if lines.size() == 1:
		lines.append("No damage-related components found")
	return "\n".join(lines)
```

```gdscript
	if not file:
		return "Error: failed to open script file"
	var source_code := file.get_as_text()
	file.close()
	if source_code.is_empty():
		return "Error: script file is empty"
	var gd := GDScript.new()
	gd.set_source_code(source_code)
	var err := gd.reload()

		DirAccess.remove_absolute(SCRIPT_FILE)
		return "Error: failed to compile script (error %d)" % err
	var instance = gd.new()
	if not instance:
		DirAccess.remove_absolute(SCRIPT_FILE)
		return "Error: failed to instantiate script"
	if not instance.has_method("run"):
		instance.free()
		DirAccess.remove_absolute(SCRIPT_FILE)
		return "Error: script must have 'run()' method"
	var result = instance.run()
	instance.free()
	DirAccess.remove_absolute(SCRIPT_FILE)
	return "Script result: " + str(result)
```

```gdscript
func _apply_completion() -> void:
	if _completions.size() == 0 or _completion_index >= _completions.size():
		return

	var console := _get_console()
	if console == null:
		return

	var state: Dictionary = console.describe_cursor(_input, _cursor_index)
	var prefix: String = String(state.get("prefix", ""))
	var selected: String = _completions[_completion_index]

	# Replace the in-progress token with the selected completion, preserve
	# everything typed before it, append a trailing space so the user can
	# continue typing the next slot.
	var left := _input.substr(0, _cursor_index)
	if prefix.length() > 0:
		left = left.substr(0, left.length() - prefix.length())
	var right := _input.substr(_cursor_index)
	_input = left + selected + " " + right
	_cursor_index = (left + selected + " ").length()

	_show_completions = false
	_user_navigated_completions = false
	_update_completions()
```
