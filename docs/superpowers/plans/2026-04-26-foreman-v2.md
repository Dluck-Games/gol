# Foreman v2 Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Rewrite Foreman from a multi-agent daemon orchestrator into a simple CLI task execution engine that spawns a single coding agent per task.

**Architecture:** CLI entry point (`bin/foreman.mts`) routes commands to 7 library modules. YAML task templates define what to run. Runner spawns coding agents with env-file-driven client switching. State tracked in a single JSON file. Worktrees provide branch isolation. Pluggable notifier for completion alerts.

**Tech Stack:** TypeScript ESM (`.mts`), `tsx` runtime, `yaml` package for YAML parsing, `node:util/parseArgs` for CLI, no build step.

**Spec:** `docs/superpowers/specs/2026-04-26-foreman-v2-design.md`

```typescript
import { existsSync, mkdirSync, writeFileSync, readFileSync, openSync, fsyncSync, closeSync, renameSync, unlinkSync } from 'node:fs'
import { dirname, join } from 'node:path'
import { randomBytes } from 'node:crypto'

export interface Task {
  id: string
  taskName: string
  status: 'running' | 'done' | 'failed' | 'cancelled'
  pid: number | null
  issueNumber: number | null
  prUrl: string | null
  workspace: string | null
  client: string
  logFile: string
  startedAt: string
  finishedAt: string | null
  exitCode: number | null
  notify: boolean
  notified: boolean
}

export interface ForemanState {
  version: 1
  tasks: Record<string, Task>
}

function emptyState(): ForemanState {
  return { version: 1, tasks: {} }
}

export function generateTaskId(): string {
  const date = new Date().toISOString().slice(0, 10).replace(/-/g, '')
  const hex = randomBytes(2).toString('hex')
  return `task_${date}_${hex}`
}

export function readState(statePath: string): ForemanState {
  if (!existsSync(statePath)) return emptyState()
  try {
    const raw = readFileSync(statePath, 'utf-8')
    return JSON.parse(raw) as ForemanState
  } catch {
    return emptyState()
  }
}

export function writeState(statePath: string, state: ForemanState): void {
  const dir = dirname(statePath)
  if (!existsSync(dir)) mkdirSync(dir, { recursive: true })

  const tmpPath = statePath + '.tmp'
  const data = JSON.stringify(state, null, 2) + '\n'

  writeFileSync(tmpPath, data, 'utf-8')

  let fd: number | undefined
  try {
    fd = openSync(tmpPath, 'r')
    fsyncSync(fd)
  } finally {
    if (fd !== undefined) closeSync(fd)
  }

  renameSync(tmpPath, statePath)
}

export function createTask(statePath: string, opts: {
  id: string
  taskName: string
  client: string
  logFile: string
  pid: number | null
  issueNumber?: number | null
  workspace?: string | null
  notify: boolean
}): Task {
  const state = readState(statePath)
  const task: Task = {
    id: opts.id,
    taskName: opts.taskName,
    status: 'running',
    pid: opts.pid,
    issueNumber: opts.issueNumber ?? null,
    prUrl: null,
    workspace: opts.workspace ?? null,
    client: opts.client,
    logFile: opts.logFile,
    startedAt: new Date().toISOString(),
    finishedAt: null,
    exitCode: null,
    notify: opts.notify,
    notified: false,
  }
  state.tasks[task.id] = task
  writeState(statePath, state)
  return task
}

export function updateTask(statePath: string, taskId: string, updates: Partial<Task>): void {
  const state = readState(statePath)
  const task = state.tasks[taskId]
  if (!task) return
  Object.assign(task, updates)
  writeState(statePath, state)
}

export function getTask(statePath: string, taskId: string): Task | undefined {
  return readState(statePath).tasks[taskId]
}

export function getAllTasks(statePath: string): Task[] {
  return Object.values(readState(statePath).tasks)
}
```

```typescript
import { readdirSync, readFileSync, existsSync } from 'node:fs'
import { join, basename } from 'node:path'
import { parse as parseYaml } from 'yaml'

export interface TaskArg {
  name: string
  required: boolean
  description: string
}

export interface TaskTemplate {
  name: string
  description: string
  client: string
  skills?: string[]
  args?: TaskArg[]
  prompt?: string
  status?: 'placeholder'
}

export function loadTasks(tasksDir: string): Map<string, TaskTemplate> {
  const tasks = new Map<string, TaskTemplate>()
  if (!existsSync(tasksDir)) return tasks

  for (const file of readdirSync(tasksDir)) {
    if (!file.endsWith('.yaml') && !file.endsWith('.yml')) continue
    const raw = readFileSync(join(tasksDir, file), 'utf-8')
    const template = parseYaml(raw) as TaskTemplate
    if (!template.name || !template.description) {
      console.error(`Warning: skipping invalid task template ${file} (missing name or description)`)
      continue
    }
    tasks.set(template.name, template)
  }
  return tasks
}

export function getTask(tasks: Map<string, TaskTemplate>, name: string): TaskTemplate {
  const task = tasks.get(name)
  if (!task) {
    const available = [...tasks.keys()].join(', ')
    throw new Error(`Unknown task '${name}'. Available: ${available}`)
  }
  if (task.status === 'placeholder') {
    throw new Error(`Task '${name}' is not yet implemented (placeholder only)`)
  }
  if (!task.prompt) {
    throw new Error(`Task '${name}' has no prompt defined`)
  }
  return task
}

export function validateArgs(task: TaskTemplate, providedArgs: Record<string, string>): void {
  if (!task.args) return
  for (const arg of task.args) {
    if (arg.required && !providedArgs[arg.name]) {
      throw new Error(`Missing required argument '--${arg.name}' for task '${task.name}'`)
    }
  }
}

export function buildSkillsInstruction(skills: string[] | undefined): string {
  if (!skills || skills.length === 0) return ''
  const list = skills.map(s => `- ${s}`).join('\n')
  return `Use the Skill tool to invoke the following skills, then follow each exactly:\n${list}`
}

export function renderPrompt(task: TaskTemplate, vars: Record<string, string>): string {
  if (!task.prompt) throw new Error(`Task '${task.name}' has no prompt`)
  let prompt = task.prompt
  for (const [key, value] of Object.entries(vars)) {
    prompt = prompt.replaceAll(`{{${key}}}`, value)
  }
  // Remove any remaining unreplaced variables
  prompt = prompt.replace(/\{\{[a-z_]+\}\}/g, '')
  return prompt.trim()
}
```

```typescript
import { execFile } from 'node:child_process'

function gh(args: string[], repo: string): Promise<string> {
  return new Promise((resolve, reject) => {
    execFile('gh', [...args, '-R', repo], { timeout: 30_000 }, (err, stdout, stderr) => {
      if (err) {
        reject(new Error(`gh ${args[0]} failed: ${stderr || err.message}`))
        return
      }
      resolve(stdout.trim())
    })
  })
}

export async function getIssueBody(repo: string, issueNumber: string | number): Promise<string> {
  try {
    return await gh(['issue', 'view', String(issueNumber), '--json', 'body', '--jq', '.body'], repo)
  } catch (e) {
    throw new Error(`Failed to fetch issue #${issueNumber}: ${(e as Error).message}`)
  }
}
```

```typescript
import { existsSync, rmSync } from 'node:fs'
import { execFileSync } from 'node:child_process'
import { join, basename } from 'node:path'
import { randomBytes } from 'node:crypto'

export interface WorktreeResult {
  path: string
  branch: string
}

function timestamp(): string {
  return new Date().toISOString().replace(/[-:T]/g, '').slice(0, 15)
}

function prune(repoDir: string): void {
  try {
    execFileSync('git', ['-C', repoDir, 'worktree', 'prune'], { timeout: 30_000, stdio: 'pipe' })
  } catch {
    // non-fatal
  }
}

export function createWorktree(repoDir: string, wsDir: string, branchName: string): WorktreeResult {
  const wsName = `ws_${timestamp()}_${randomBytes(2).toString('hex')}`
  const wsPath = join(wsDir, wsName)

  prune(repoDir)

  try {
    execFileSync('git', ['-C', repoDir, 'fetch', 'origin', 'main'], { timeout: 120_000, stdio: 'pipe' })
  } catch {
    // proceed with local state
  }

  try {
    execFileSync(
      'git', ['-C', repoDir, 'worktree', 'add', '-b', branchName, wsPath, 'origin/main'],
      { timeout: 30_000, stdio: 'pipe' }
    )
  } catch {
    // branch might already exist — try checking it out
    execFileSync(
      'git', ['-C', repoDir, 'worktree', 'add', wsPath, branchName],
      { timeout: 30_000, stdio: 'pipe' }
    )
  }

  return { path: wsPath, branch: branchName }
}

export function destroyWorktree(repoDir: string, wsPath: string, wsDir: string): void {
  if (!wsPath || !existsSync(wsPath)) return
  if (!wsPath.startsWith(wsDir)) {
    throw new Error(`Refusing to destroy path outside workspace dir: ${wsPath}`)
  }

  try {
    execFileSync('git', ['-C', repoDir, 'worktree', 'remove', wsPath, '--force'], {
      timeout: 30_000, stdio: 'pipe',
    })
  } catch {
    rmSync(wsPath, { recursive: true, force: true })
  }

  prune(repoDir)
}

export function buildBranchName(taskName: string, issueNumber?: number | null): string {
  if (taskName === 'fix-issue' && issueNumber) {
    return `foreman/issue-${issueNumber}`
  }
  const ts = timestamp().slice(0, 8)
  return `foreman/${taskName}-${ts}`
}
```

```typescript
import { execFile } from 'node:child_process'

export interface TaskResult {
  taskId: string
  taskName: string
  status: 'done' | 'failed' | 'cancelled'
  issueNumber: number | null
  prUrl: string | null
  duration: string
  summary: string
}

export interface Notifier {
  name: string
  notify(result: TaskResult): Promise<void>
}

export interface NotifyConfig {
  default: string
  openclaw?: { target: string; channel: string }
  system?: Record<string, never>
}

function formatMessage(result: TaskResult): string {
  const icon = result.status === 'done' ? '✅' : result.status === 'failed' ? '❌' : '⚪'
  let msg = `${icon} Foreman: ${result.taskName} ${result.status}`
  if (result.issueNumber) msg += ` (#${result.issueNumber})`
  msg += ` [${result.duration}]`
  if (result.prUrl) msg += `\nPR: ${result.prUrl}`
  if (result.summary) msg += `\n${result.summary}`
  return msg
}

function createOpenclawNotifier(config: { target: string; channel: string }): Notifier {
  return {
    name: 'openclaw',
    async notify(result) {
      const message = formatMessage(result)
      return new Promise<void>((resolve, reject) => {
        execFile('openclaw', [
          'message', 'send',
          '--target', config.target,
          '--channel', config.channel,
          '--message', message,
        ], { timeout: 15_000 }, (err) => {
          if (err) reject(new Error(`openclaw notify failed: ${err.message}`))
          else resolve()
        })
      })
    },
  }
}

function createSystemNotifier(): Notifier {
  return {
    name: 'system',
    async notify(result) {
      const message = formatMessage(result)
      return new Promise<void>((resolve, reject) => {
        execFile('osascript', [
          '-e', `display notification "${message.replace(/"/g, '\\"')}" with title "Foreman"`,
        ], { timeout: 5_000 }, (err) => {
          if (err) reject(new Error(`system notify failed: ${err.message}`))
          else resolve()
        })
      })
    },
  }
}

export function createNotifier(config: NotifyConfig): Notifier {
  const backend = config.default
  if (backend === 'openclaw' && config.openclaw) {
    return createOpenclawNotifier(config.openclaw)
  }
  if (backend === 'system') {
    return createSystemNotifier()
  }
  throw new Error(`Unknown notify backend: ${backend}`)
}
```

```typescript
import { spawn, execFileSync } from 'node:child_process'
import { createWriteStream, readFileSync, existsSync, mkdirSync } from 'node:fs'
import { join, dirname } from 'node:path'
import { updateTask } from './state.mts'
import { createNotifier, type NotifyConfig, type TaskResult } from './notifier.mts'
import { destroyWorktree } from './worktree.mts'

export interface RunOptions {
  taskId: string
  taskName: string
  client: string
  binary: string
  envFile?: string
  prompt: string
  detach: boolean
  notify: boolean
  notifyConfig?: NotifyConfig
  workDir: string
  repoDir: string
  wsDir: string
  workspace: string | null
  logFile: string
  statePath: string
  issueNumber: number | null
  foremanBin: string
}

function parseEnvFile(filePath: string): Record<string, string> {
  if (!existsSync(filePath)) return {}
  const content = readFileSync(filePath, 'utf-8')
  const env: Record<string, string> = {}
  for (const line of content.split('\n')) {
    const trimmed = line.trim()
    if (!trimmed || trimmed.startsWith('#')) continue
    const eqIdx = trimmed.indexOf('=')
    if (eqIdx === -1) continue
    env[trimmed.slice(0, eqIdx)] = trimmed.slice(eqIdx + 1)
  }
  return env
}

function buildEnv(envFile: string | undefined, foremanDir: string): NodeJS.ProcessEnv {
  const base = { ...process.env }
  if (envFile) {
    const envPath = join(foremanDir, envFile)
    const fileEnv = parseEnvFile(envPath)
    Object.assign(base, fileEnv)
  }
  return base
}

function buildClientArgs(binary: string, prompt: string): string[] {
  // Claude Code (claude binary)
  if (binary === 'claude') {
    return ['-p', prompt, '--output-format', 'stream-json', '--verbose']
  }
  // OpenCode — headless mode
  if (binary === 'opencode') {
    return ['--prompt', prompt]
  }
  return [prompt]
}

function computeDuration(startedAt: string): string {
  const ms = Date.now() - new Date(startedAt).getTime()
  const secs = Math.floor(ms / 1000)
  if (secs < 60) return `${secs}s`
  const mins = Math.floor(secs / 60)
  const remSecs = secs % 60
  if (mins < 60) return `${mins}m ${remSecs}s`
  const hrs = Math.floor(mins / 60)
  const remMins = mins % 60
  return `${hrs}h ${remMins}m`
}

async function onTaskComplete(opts: RunOptions, exitCode: number, startedAt: string): Promise<void> {
  const status = exitCode === 0 ? 'done' as const : 'failed' as const
  updateTask(opts.statePath, opts.taskId, {
    status,
    finishedAt: new Date().toISOString(),
    exitCode,
    pid: null,
  })

  if (opts.workspace) {
    destroyWorktree(opts.repoDir, opts.workspace, opts.wsDir)
  }

  if (opts.notify && opts.notifyConfig) {
    const task = (await import('./state.mts')).getTask(opts.statePath, opts.taskId)
    if (task && !task.notified) {
      try {
        const notifier = createNotifier(opts.notifyConfig)
        const result: TaskResult = {
          taskId: opts.taskId,
          taskName: opts.taskName,
          status,
          issueNumber: opts.issueNumber,
          prUrl: task.prUrl,
          duration: computeDuration(startedAt),
          summary: `Exit code: ${exitCode}`,
        }
        await notifier.notify(result)
        updateTask(opts.statePath, opts.taskId, { notified: true })
      } catch (e) {
        console.error(`Notification failed: ${(e as Error).message}`)
      }
    }
  }
}

export function runForeground(opts: RunOptions): Promise<number> {
  const foremanDir = dirname(dirname(new URL(import.meta.url).pathname))
  const env = buildEnv(opts.envFile, foremanDir)
  const args = buildClientArgs(opts.binary, opts.prompt)

  const logDir = dirname(opts.logFile)
  if (!existsSync(logDir)) mkdirSync(logDir, { recursive: true })
  const logStream = createWriteStream(opts.logFile, { flags: 'a' })

  const startedAt = new Date().toISOString()
  const child = spawn(opts.binary, args, {
    cwd: opts.workDir,
    env,
    stdio: ['ignore', 'pipe', 'pipe'],
    detached: true,
  })

  updateTask(opts.statePath, opts.taskId, { pid: child.pid ?? null })

  child.stdout?.on('data', (chunk: Buffer) => {
    process.stdout.write(chunk)
    logStream.write(chunk)
  })
  child.stderr?.on('data', (chunk: Buffer) => {
    process.stderr.write(chunk)
    logStream.write(chunk)
  })

  return new Promise<number>((resolve) => {
    child.on('exit', async (code) => {
      logStream.end()
      const exitCode = code ?? 1
      await onTaskComplete(opts, exitCode, startedAt)
      resolve(exitCode)
    })
  })
}

export function runDetached(opts: RunOptions): void {
  const foremanDir = dirname(dirname(new URL(import.meta.url).pathname))
  const wrapperPath = join(foremanDir, 'lib', 'detach-wrapper.mts')

  const wrapperArgs = [
    wrapperPath,
    '--task-id', opts.taskId,
    '--state-path', opts.statePath,
    '--binary', opts.binary,
    '--log-file', opts.logFile,
    '--work-dir', opts.workDir,
    '--task-name', opts.taskName,
    '--repo-dir', opts.repoDir,
    '--ws-dir', opts.wsDir,
  ]
  if (opts.envFile) wrapperArgs.push('--env-file', opts.envFile)
  if (opts.workspace) wrapperArgs.push('--workspace', opts.workspace)
  if (opts.notify) wrapperArgs.push('--notify')
  if (opts.notifyConfig) wrapperArgs.push('--notify-config', JSON.stringify(opts.notifyConfig))
  if (opts.issueNumber) wrapperArgs.push('--issue-number', String(opts.issueNumber))
  wrapperArgs.push('--', opts.prompt)

  const logDir = dirname(opts.logFile)
  if (!existsSync(logDir)) mkdirSync(logDir, { recursive: true })

  const child = spawn('tsx', wrapperArgs, {
    cwd: opts.workDir,
    stdio: 'ignore',
    detached: true,
  })

  child.unref()

  updateTask(opts.statePath, opts.taskId, { pid: child.pid ?? null })
}
```

```typescript
import { existsSync, readFileSync, writeFileSync, mkdirSync } from 'node:fs'
import { execSync } from 'node:child_process'
import { dirname } from 'node:path'
import { randomBytes } from 'node:crypto'

export interface CronEntry {
  id: string
  task: string
  schedule: string
  prompt?: string
  createdAt: string
}

const CRON_MARKER = '# foreman-cron:'

function cronFilePath(dataDir: string): string {
  return `${dataDir}/cron-entries.json`
}

export function readCronEntries(dataDir: string): CronEntry[] {
  const path = cronFilePath(dataDir)
  if (!existsSync(path)) return []
  try {
    return JSON.parse(readFileSync(path, 'utf-8'))
  } catch {
    return []
  }
}

function writeCronEntries(dataDir: string, entries: CronEntry[]): void {
  const path = cronFilePath(dataDir)
  const dir = dirname(path)
  if (!existsSync(dir)) mkdirSync(dir, { recursive: true })
  writeFileSync(path, JSON.stringify(entries, null, 2) + '\n', 'utf-8')
}

function getCurrentCrontab(): string {
  try {
    return execSync('crontab -l 2>/dev/null', { encoding: 'utf-8' })
  } catch {
    return ''
  }
}

function setCrontab(content: string): void {
  execSync(`echo ${JSON.stringify(content)} | crontab -`, { encoding: 'utf-8' })
}

export function addCronEntry(
  dataDir: string,
  foremanBin: string,
  task: string,
  schedule: string,
  prompt?: string,
): CronEntry {
  const entries = readCronEntries(dataDir)
  const id = `cron_${new Date().toISOString().slice(0, 10).replace(/-/g, '')}_${randomBytes(2).toString('hex')}`

  const entry: CronEntry = {
    id,
    task,
    schedule,
    prompt,
    createdAt: new Date().toISOString(),
  }
  entries.push(entry)
  writeCronEntries(dataDir, entries)

  // Install crontab line
  let cmd = `${foremanBin} run task ${task} --detach`
  if (prompt) cmd += ` --prompt ${JSON.stringify(prompt)}`

  const crontab = getCurrentCrontab()
  const newLine = `${schedule} ${cmd} ${CRON_MARKER}${id}`
  setCrontab(crontab.trimEnd() + '\n' + newLine + '\n')

  return entry
}

export function removeCronEntry(dataDir: string, cronId: string): void {
  const entries = readCronEntries(dataDir)
  const filtered = entries.filter(e => e.id !== cronId)
  if (filtered.length === entries.length) {
    throw new Error(`Cron entry '${cronId}' not found`)
  }
  writeCronEntries(dataDir, filtered)

  // Remove from crontab
  const crontab = getCurrentCrontab()
  const lines = crontab.split('\n').filter(line => !line.includes(`${CRON_MARKER}${cronId}`))
  setCrontab(lines.join('\n'))
}

export function listCronEntries(dataDir: string): Array<CronEntry & { active: boolean }> {
  const entries = readCronEntries(dataDir)
  const crontab = getCurrentCrontab()
  return entries.map(entry => ({
    ...entry,
    active: crontab.includes(`${CRON_MARKER}${entry.id}`),
  }))
}
```

```typescript
// --- Pre-run validation (fast doctor subset) ---

function preRunCheck(config: ForemanConfig, clientName: string, taskSkills: string[] | undefined, workDir: string): void {
  const clientConfig = config.clients[clientName]
  if (!clientConfig) throw new Error(`Unknown client '${clientName}'. Available: ${Object.keys(config.clients).join(', ')}`)

  // Check binary on PATH
  try {
    const { execFileSync } = require('node:child_process')
    execFileSync('which', [clientConfig.binary], { stdio: 'pipe' })
  } catch {
    throw new Error(`Client binary '${clientConfig.binary}' not found on PATH`)
  }

  // Check env file
  if (clientConfig.envFile) {
    const envPath = join(foremanDir, clientConfig.envFile)
    if (!existsSync(envPath)) throw new Error(`Env file not found: ${envPath}`)
  }

  // Check skills
  if (taskSkills) {
    for (const skill of taskSkills) {
      if (skill.includes(':')) continue // plugin skill, skip file check
      const skillPath = join(workDir, '.claude', 'skills', skill, 'SKILL.md')
      if (!existsSync(skillPath)) {
        throw new Error(`Skill '${skill}' not found at ${skillPath}`)
      }
    }
  }
}

  }

  const template = getTaskTemplate(tasks, taskName)

  // Parse remaining args dynamically based on task's declared args
  const optionDefs: Record<string, { type: 'string' | 'boolean' }> = {
    prompt: { type: 'string' },
    client: { type: 'string' },
    notify: { type: 'boolean' },
    detach: { type: 'boolean' },
  }
  if (template.args) {
    for (const arg of template.args) {
      optionDefs[arg.name] = { type: 'string' }
    }
  }

  const { values: flags } = parseArgs({
    args: args.slice(1),
    options: optionDefs,
    strict: false,
  })

  const clientName = (flags.client as string) || template.client || 'opencode'
  const extraPrompt = (flags.prompt as string) || ''
  const shouldNotify = !!flags.notify
  const shouldDetach = !!flags.detach

  // Collect task-specific args
  const taskArgs: Record<string, string> = {}
  if (template.args) {
    for (const arg of template.args) {
      if (flags[arg.name]) taskArgs[arg.name] = flags[arg.name] as string
    }
  }
  validateArgs(template, taskArgs)

  // Pre-run check
  const clientConfig = config.clients[clientName]
  preRunCheck(config, clientName, template.skills, workDir)

  // Generate task ID
  const taskId = generateTaskId()
  const logFile = resolveLogFile(workDir, taskId)

  // Determine issue number
  const issueNumber = taskArgs.issue ? parseInt(taskArgs.issue, 10) : null

  // Fetch issue body if needed
  let issueBody = ''
  if (issueNumber) {
    try {
      issueBody = await getIssueBody(config.repo, issueNumber)
    } catch (e) {
      console.error(`Warning: ${(e as Error).message}`)
    }
  }

  // Create worktree
  const branchName = buildBranchName(taskName, issueNumber)
  const worktree = createWorktree(repoDir, wsDir, branchName)

  // Build notify instruction
  let notifyInstruction = ''
  if (shouldNotify) {
    const foremanBin = resolve(foremanDir, 'bin', 'foreman.mts')
    notifyInstruction = `When you finish this task, run this command to send a completion notification:\ntsx ${foremanBin} notify --task-id ${taskId} --status <done|failed> --summary "brief summary of what was done"`
  }

  // Build worktree instruction
  const worktreeInstruction = `A worktree has been created at ${worktree.path} on branch ${worktree.branch}. Do your coding work there. Do not modify the main working directory.`

  // Render prompt
  const vars: Record<string, string> = {
    ...taskArgs,
    skills_instruction: buildSkillsInstruction(template.skills),
    issue_body: issueBody,
    extra_prompt: extraPrompt,
    notify_instruction: notifyInstruction,
    worktree_instruction: worktreeInstruction,
    workspace_path: worktree.path,
    branch_name: worktree.branch,
  }
  const prompt = renderPrompt(template, vars)

  // Create task in state
  createTask(statePath, {
    id: taskId,
    taskName,
    client: clientName,
    logFile,
    pid: null,
    issueNumber,
    workspace: worktree.path,
    notify: shouldNotify,
  })

  console.log(`Task ${taskId} started: ${taskName} (client: ${clientName})`)
  if (issueNumber) console.log(`Issue: #${issueNumber}`)
  console.log(`Worktree: ${worktree.path}`)
  console.log(`Log: ${logFile}`)

  const runOpts = {
    taskId,
    taskName,
    client: clientName,
    binary: clientConfig.binary,
    envFile: clientConfig.envFile,
    prompt,
    detach: shouldDetach,
    notify: shouldNotify,
    notifyConfig: shouldNotify ? config.notify : undefined,
    workDir,
    repoDir,
    wsDir,
    workspace: worktree.path,
    logFile,
    statePath,
    issueNumber,
    foremanBin: resolve(foremanDir, 'bin', 'foreman.mts'),
  }

  if (shouldDetach) {
    runDetached(runOpts)
    console.log(`Task ${taskId} detached (PID will be in state file)`)
    return 0
  }

  return runForeground(runOpts)
}

      ? formatDuration(new Date(task.startedAt), new Date(task.finishedAt))
      : formatDuration(new Date(task.startedAt), new Date())
    const pr = task.prUrl || ''
    console.log(
      `${task.id.padEnd(22)}${task.taskName.padEnd(22)}${task.status.padEnd(11)}${task.client.padEnd(11)}${duration.padEnd(11)}${pr}`

function formatDuration(start: Date, end: Date): string {
  const ms = end.getTime() - start.getTime()
  const secs = Math.floor(ms / 1000)
  if (secs < 60) return `${secs}s`
  const mins = Math.floor(secs / 60)
  if (mins < 60) return `${mins}m`
  const hrs = Math.floor(mins / 60)
  return `${hrs}h ${mins % 60}m`
}

async function handleNotify(args: string[]): Promise<number> {
  const { values: flags } = parseArgs({
    args,
    options: {
      'task-id': { type: 'string' },
      status: { type: 'string' },
      summary: { type: 'string' },

    strict: false,
  })
  if (!flags['task-id'] || !flags.status) {

    return 1
  }
  const config = loadConfig()
  const workDir = resolveWorkDir()
  const statePath = resolveStatePath(workDir)
  const task = getTaskState(statePath, flags['task-id'])

  return 0

function handleDoctor(): number {
  const config = loadConfig()
  const tasks = loadTasks(tasksDir)
  const workDir = resolveWorkDir()
  let ok = true

  // Check client binaries
  for (const [name, client] of Object.entries(config.clients)) {
    try {
      const { execFileSync } = require('node:child_process')
      execFileSync('which', [client.binary], { stdio: 'pipe' })
      console.log(`✓ Client '${name}' found (${client.binary})`)
    } catch {
      console.log(`✗ Client '${name}' binary '${client.binary}' not found on PATH`)
      ok = false
    }

    if (client.envFile) {
      const envPath = join(foremanDir, client.envFile)
      if (existsSync(envPath)) {
        console.log(`✓ Env file '${client.envFile}' exists`)
      } else {
        console.log(`✗ Env file '${client.envFile}' not found at ${envPath}`)
        ok = false
      }
    }
  }

  // Check skills
  for (const [, task] of tasks) {
    if (!task.skills) continue
    for (const skill of task.skills) {
      if (skill.includes(':')) {
        console.log(`⊘ Skill '${skill}' is a plugin skill — verify plugin is installed`)
        continue
      }
      const skillPath = join(workDir, '.claude', 'skills', skill, 'SKILL.md')
      if (existsSync(skillPath)) {
        console.log(`✓ Skill '${skill}' exists`)
      } else {
        console.log(`✗ Skill '${skill}' not found at ${skillPath}`)
        ok = false
      }
    }
  }

  // Check task templates
  for (const [, task] of tasks) {
    if (task.status === 'placeholder') {
      console.log(`⊘ Task '${task.name}' is a placeholder`)
    } else {
      console.log(`✓ Task '${task.name}' valid`)
    }
  }

  // Check config
  console.log(`✓ Config valid`)

  // Check state directory
  const stateDir = join(workDir, '.foreman')
  if (existsSync(stateDir)) {
    console.log(`✓ State directory exists`)
  } else {
    console.log(`⊘ State directory will be created on first run`)
  }

  // Check git repo
  const repoDir = resolveRepoDir(workDir)
  if (existsSync(join(repoDir, '.git'))) {
    console.log(`✓ Git repo OK (${repoDir})`)
  } else {
    console.log(`✗ Git repo not found at ${repoDir}`)
    ok = false
  }

  // Check gh CLI
  try {
    const { execFileSync } = require('node:child_process')
    const result = execFileSync('gh', ['auth', 'status'], { stdio: 'pipe', encoding: 'utf-8' })
    console.log(`✓ gh authenticated`)
  } catch {
    console.log(`✗ gh not authenticated (run 'gh auth login')`)
    ok = false
  }

  return ok ? 0 : 1
}

      return handleNotify(args.slice(1))

      return handleDoctor()
```
